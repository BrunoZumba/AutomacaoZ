CREATE SEQUENCE public.aztb030_actionbutton_id_actionbutton_seq;

CREATE TABLE public.AZTB030_actionButton (
                id_actionButton INTEGER NOT NULL DEFAULT nextval('public.aztb030_actionbutton_id_actionbutton_seq'),
                no_action VARCHAR(50) NOT NULL,
                seq_acao INTEGER NOT NULL,
                json_actoin VARCHAR NOT NULL,
                CONSTRAINT id_actionbutton PRIMARY KEY (id_actionButton, no_action, seq_acao)
);
COMMENT ON TABLE public.AZTB030_actionButton IS 'Dados de um ActionButton';
COMMENT ON COLUMN public.AZTB030_actionButton.id_actionButton IS 'Identificador único do Action Button';
COMMENT ON COLUMN public.AZTB030_actionButton.seq_acao IS 'Sequencia da ação dentro do action button';
COMMENT ON COLUMN public.AZTB030_actionButton.json_actoin IS 'Json contendo cada ação, identificada pelo nome e o sequencial';


ALTER SEQUENCE public.aztb030_actionbutton_id_actionbutton_seq OWNED BY public.AZTB030_actionButton.id_actionButton;

CREATE INDEX aztb030_actionbutton_idx
 ON public.AZTB030_actionButton
 ( no_action ASC, seq_acao ASC );
